import numpy as np
from tensorflow.keras.preprocessing.text import Tokenizer
from tensorflow.keras.preprocessing.sequence import pad_sequences
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Embedding,LSTM,Dense
from tensorflow.keras.utils import to_categorical

corpus=[
"how are you zahid","how is your day","i am fine zahid",
"i am going zahid home","how are things",
"are you zahid coming today","i will call you"
]

tok=Tokenizer(); tok.fit_on_texts(corpus)
v=len(tok.word_index)+1

seq=[]
for s in corpus:
    t=tok.texts_to_sequences([s])[0]
    for i in range(1,len(t)): seq.append(t[:i+1])

m=max(len(x) for x in seq)
seq=pad_sequences(seq,maxlen=m,padding="pre")

X,y=seq[:,:-1],to_categorical(seq[:,-1],v)

model=Sequential([Embedding(v,32,input_length=m-1),LSTM(64),Dense(v,activation="softmax")])
model.compile(loss="categorical_crossentropy",optimizer="adam")
model.fit(X,y,epochs=200,verbose=0)

def predict(t):
    s=pad_sequences([tok.texts_to_sequences([t])[0]],maxlen=m-1,padding="pre")
    p=np.argmax(model.predict(s,verbose=0))
    for w,i in tok.word_index.items():
        if i==p: return w

for t in ["how are you","i am fine","are you"]:
    print(t,"->",predict(t))
